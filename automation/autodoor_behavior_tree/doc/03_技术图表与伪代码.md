**防止假卡死的关键设计：**

| 组件 | 作用 | 注意事项 |
|------|------|---------|
| `_task_queue` | 线程安全队列，存储待处理的 UI 更新任务 | 使用 `Queue`，不使用 `list` |
| `_poll()` | 主线程轮询，自适应间隔（运行10ms/空闲100ms） | **必须在编辑器初始化时启动** |
| `_poll_daemon()` | 后台线程守护，当 `_widget` 为 None 时继续尝试 | 防止轮询永久停止 |
| `_node_status_cache` | 合并相同节点的重复状态更新 | 减少不必要的重绘 |

### IbInputSimulator 操作流程

```
键盘操作:
  Hook ON → SendInput(KEYBDINPUT) → Hook OFF

鼠标移动:
  Hook ON → SetCursorPos(x, y) → SendInput(MOVE|ABSOLUTE, norm_x, norm_y) → Hook OFF

鼠标点击:
  Hook ON → SetCursorPos(x, y) → SendInput(MOVE|ABSOLUTE) → SendInput(LDOWN) → sleep → SendInput(LUP) → Hook OFF

鼠标滚轮:
  Hook ON → SendInput(WHEEL, delta) → Hook OFF
```

### 坐标浮动应用流程

```
1. 获取基础坐标 position (可能经过窗口→屏幕坐标转换)
2. if x_float > 0: actual_x = random(position.x - x_float, position.x + x_float)
3. if y_float > 0: actual_y = random(position.y - y_float, position.y + y_float)
4. 使用 (actual_x, actual_y) 作为最终坐标
```

```
Input: hwnd (Optional[int]), pid (Optional[int]), title_keyword (str)
Output: Tuple[Optional[int], str] - (窗口句柄, 查找方式)

function find_window_smart_with_hwnd(hwnd, pid, title_keyword):
    // 第一优先级：直接使用句柄
    if hwnd and is_window_valid(hwnd):
        return (hwnd, "hwnd")
    
    // 第二优先级：通过PID智能查找
    if pid:
        found_hwnd = find_main_window_by_pid(pid)
        if found_hwnd:
            return (found_hwnd, "pid")
    
    // 第三优先级：通过标题关键字查找
    if title_keyword:
        found_hwnd = find_window_by_title(title_keyword)
        if found_hwnd:
            return (found_hwnd, "title")
    
    return (None, "not_found")


Algorithm: SmartSwitchToBoundWindow
Input: None
Output: bool

function smart_switch_to_bound_window():
    if not self._bound_window:
        return False
    if WindowManager.is_foreground_window(self._bound_window):
        return True
    self._saved_foreground_window = WindowManager.get_foreground_window()
    return WindowManager.switch_to_window(self._bound_window)
```

```
function get_random_value(base_value, random_range=0, min_value=null):
    if random_range <= 0:
        return base_value
    min_val = base_value - random_range
    max_val = base_value + random_range
    if min_value is not None:
        min_val = max(min_val, min_value)
    return random.randint(int(min_val), int(max_val))
```

### 9.3 Parallel 节点 Tick 算法

```
Algorithm: ParallelTick
Input: children[], cached_statuses, success_policy, context
Output: NodeStatus

function parallel_tick(children, cached_statuses, success_policy, context):
    if children is empty:
        return SUCCESS
    
    success_count = 0
    failure_count = 0
    running_children = []
    enabled_count = 0
    
    for i, child in enumerate(children):
        if not child.config.enabled:
            continue
        enabled_count += 1
        
        if i in cached_statuses:
            status = cached_statuses[i]
            if status == SUCCESS: success_count += 1; continue
            if status == FAILURE: failure_count += 1; continue
        
        status = child.tick(context)
        
        switch status:
            case SUCCESS: cached_statuses[i] = SUCCESS; success_count += 1
            case FAILURE: cached_statuses[i] = FAILURE; failure_count += 1
            case RUNNING: running_children.append(child)
    
    // require_one 策略：任一成功即返回
    if success_policy == REQUIRE_ONE and success_count > 0:
        abort_running(running_children)
        return SUCCESS
    
    if running_children is not empty:
        return RUNNING
    
    // 所有子节点执行完毕
    if success_policy == REQUIRE_ALL:
        return SUCCESS if success_count == enabled_count else FAILURE
    else:
        return SUCCESS if success_count > 0 else FAILURE
```

```
function find_template(source, template, threshold):
    source_cv = cv2.cvtColor(np.array(source), cv2.COLOR_RGB2BGR)
    template_cv = cv2.cvtColor(np.array(template), cv2.COLOR_RGB2BGR)
    
    result = cv2.matchTemplate(source_cv, template_cv, cv2.TM_CCOEFF_NORMED)
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)
    
    if max_val >= threshold:
        h, w = template_cv.shape[:2]
        center_x = max_loc[0] + w // 2
        center_y = max_loc[1] + h // 2
        return (True, (center_x, center_y), max_val)
    
    return (False, None, max_val)
```

```
        redo_stack.push(command)
        return true
    undo_stack.push(command)
    return false

    if redo_stack is empty: return false

    if command.redo():
        undo_stack.push(command)
        return true
    redo_stack.push(command)
    return false
```

```
function deserialize(data):
    nodes_data = data["nodes"]
    connections = data["connections"]
    root_id = data["root_node"]
    canvas_state = data.get("canvas_state", {})
    editor_state = data.get("editor_state", {})

    for conn in connections:
        parent = nodes[conn["parent_id"]]
        child = nodes[conn["child_id"]]
        if parent and child:
            parent.add_child(child)
    
    return (nodes[root_id], canvas_state, editor_state)
```

### 9.13 自动布局算法

```
Algorithm: AutoLayout
Input: nodes_data, root_id
Output: positions Dict[node_id → (x, y)]

const NODE_WIDTH = 180
const NODE_HEIGHT = 80
const H_GAP = 50
const V_GAP = 100

function auto_layout(nodes_data, root_id):
    positions = {}
    
    function get_children(node_id):
        return nodes_data[node_id]["children"]
    
    function calc_subtree_width(node_id):
        children = get_children(node_id)
        if children is empty:
            return NODE_WIDTH
        total = sum(calc_subtree_width(c) for c in children)
        return max(total, NODE_WIDTH)
    
    function layout_node(node_id, x, y):
        positions[node_id] = (x, y)
        children = get_children(node_id)
        if children is not empty:
            total_width = sum(calc_subtree_width(c) for c in children)
            current_x = x - total_width / 2
            for child in children:
                child_width = calc_subtree_width(child)
                layout_node(child, current_x + child_width/2, y + NODE_HEIGHT + V_GAP)
                current_x += child_width
    
    layout_node(root_id, 400, 100)
    return positions
```

## 13 OCR 管理器算法

### 13.1 图像特征分析算法

```
Algorithm: AnalyzeImageFeatures
Input: image (PIL.Image)
Output: ImageFeatures

function analyze_image_features(image):
    gray = cvt_color(image, RGB2GRAY)
    height, width = gray.shape
    
    // 亮度特征
    mean_brightness = mean(gray)
    brightness_std = std(gray)
    
    // 对比度特征（Michelson对比度）
    contrast = (max(gray) - min(gray)) / (max(gray) + min(gray) + epsilon)
    is_low_contrast = contrast < 0.3
    
    // 噪点特征（拉普拉斯方差）
    noise_level = variance(laplacian(gray)) / 10000
    has_noise = noise_level > 0.5
    
    // 字符高度估计（轮廓检测）
    binary = threshold(gray, 0, 255, THRESH_BINARY + THRESH_OTSU)
    contours = find_contours(binary, RETR_EXTERNAL)
    heights = [bounding_rect(c).height for c in contours if height > 5]
    estimated_char_height = median(heights) if heights else 20
    is_small_font = estimated_char_height < 15
    
    // 背景特征
    is_dark_background = mean_brightness < 128
    
    return ImageFeatures(
        width, height, mean_brightness, brightness_std,
        contrast, is_low_contrast, noise_level, has_noise,
        estimated_char_height, is_small_font, is_dark_background
    )
```

### 13.2 精确关键词定位算法

```
Algorithm: CalculateKeywordPosition
Input: text (string), keyword (string), box (bounding box)
Output: position (x, y)

function calculate_keyword_position(text, keyword, box):
    CHAR_WIDTH_FACTORS = {
        'chinese': 2.0, 'fullwidth': 2.0, 'uppercase': 1.2,
        'lowercase': 1.0, 'digit': 1.0, 'space': 0.5, 'punctuation': 0.6
    }
    
    keyword_idx = find(text.lower(), keyword.lower())
    if keyword_idx == -1:
        return (box.center_x, box.center_y)
    
    char_widths = [get_char_width_factor(char) for char in text]
    total_width = sum(char_widths)
    start_width = sum(char_widths[0:keyword_idx])
    keyword_width = sum(get_char_width_factor(char) for char in keyword)
    center_width = start_width + keyword_width / 2
    center_ratio = center_width / total_width
    
    x = int(box.left + box.width * center_ratio)
    y = int(box.top + box.height / 2)
    return (x, y)
```

### 13.4 识别方向排序算法

```
Algorithm: SortPositionsByDirection
Input: positions (List[(x, y)]), direction ("top-left" | "top-right" | "bottom-left" | "bottom-right")
Output: sorted_positions (List[(x, y)])

function sort_positions_by_direction(positions, direction):
    switch direction:
        case "top-left":     return sort(positions, key=lambda p: (p.y, p.x))
        case "top-right":    return sort(positions, key=lambda p: (p.y, -p.x))
        case "bottom-left":  return sort(positions, key=lambda p: (-p.y, p.x))
        case "bottom-right": return sort(positions, key=lambda p: (-p.y, -p.x))
        default:             return positions
```

### 19.3 循环检测算法伪代码

```python
def check(subtree_refs, entry_path=""):
    """DFS 循环检测"""
    # 步骤1: 归一化所有引用路径为绝对路径
    normalized_refs = []
    for ref in subtree_refs:
        abs_path = normalize_path(ref["path"], entry_path)
        normalized_refs.append({"node_id": ref["node_id"], "path": abs_path})

    # 步骤2: 构建引用关系图
    graph = {}
    for ref in normalized_refs:
        project_dir = ref["path"]
        if project_dir not in graph:
            graph[project_dir] = set()
        tree_data = load_tree_json(project_dir)
        if tree_data:
            child_refs = extract_subtree_refs(tree_data)
            for child_ref in child_refs:
                child_abs = normalize_path(child_ref["path"], project_dir)
                graph[project_dir].add(child_abs)

    # 步骤3: DFS 循环检测
    for ref in normalized_refs:
        cycle_path = detect_cycle_in_graph(graph, ref["path"])
        if cycle_path is not None:
            return (True, cycle_path)
    return (False, None)


def detect_cycle_in_graph(graph, start_node):
    """DFS 循环检测: WHITE(0) → GRAY(1) → BLACK(2)"""
    visited = {}
    path_stack = []

    def dfs(node):
        if node in visited:
            if visited[node] == 1:  # GRAY → 发现循环
                cycle_start_idx = path_stack.index(node)
                return path_stack[cycle_start_idx:] + [node]
            return None
        visited[node] = 1
        path_stack.append(node)
        for neighbor in graph.get(node, set()):
            result = dfs(neighbor)
            if result is not None:
                return result
        path_stack.pop()
        visited[node] = 2
        return None

    return dfs(start_node)
```

```python
class VariableConditionNode(ConditionNode):
    NODE_TYPE = "VariableConditionNode"

    def _check_condition(self, context) -> bool:
        var_name = self.config.get("variable_name", "")
        operator = self.config.get("operator", "==")

        if not var_name:
            return False

        # 存在性判断
        if operator == "exists":
            return context.blackboard.exists(var_name)
        if operator == "not_exists":
            return not context.blackboard.exists(var_name)

        # 获取变量当前值
        current_value = context.blackboard.get(var_name)
        if current_value is None:
            return False

        # 获取比较目标值
        compare_type = self.config.get("compare_type", "constant")
        if compare_type == "variable":
            target_value = context.blackboard.get(self.config.get("compare_variable", ""))
            if target_value is None:
                return False
        else:
            target_value = self._parse_value(self.config.get("compare_value", ""))

        return self._compare_value(current_value, operator, target_value)
```

```python
class SetVariableNode(ActionNode):
    NODE_TYPE = "SetVariableNode"
    SKIP_WINDOW_SWITCH = True

    def _execute_action(self, context) -> NodeStatus:
        var_name = self.config.get("variable_name", "")
        operation = self.config.get("operation", "set")

        if not var_name:
            return NodeStatus.FAILURE

        if operation == "set":
            value_type = self.config.get("value_type", "constant")
            if value_type == "variable":
                value = context.blackboard.get(self.config.get("source_variable", ""))
            else:
                value = self._parse_value(self.config.get("value", ""))
            context.blackboard.set(var_name, value)
            return NodeStatus.SUCCESS

        elif operation == "increment":
            current = context.blackboard.get(var_name)
            if current is None:
                return NodeStatus.FAILURE
            context.blackboard.set(var_name, current + 1)
            return NodeStatus.SUCCESS

        elif operation == "delete":
            context.blackboard.delete(var_name)
            return NodeStatus.SUCCESS
```

```python
def _check_hide_condition(self, condition: Dict) -> bool:
    depend_field = condition.get("field")
    hide_value = condition.get("value")
    if not depend_field or depend_field not in self.widgets:
        return False
    current_value = self.widgets[depend_field].get_value()
    if isinstance(hide_value, list):
        return current_value in hide_value
    return current_value == hide_value

def _update_single_field_visibility(self, key, field):
    hide_if = field.get("hide_if")
    if not hide_if:
        return
    if isinstance(hide_if, list):
        should_hide = any(self._check_hide_condition(cond) for cond in hide_if)
    else:
        should_hide = self._check_hide_condition(hide_if)
    widget = self.widgets.get(key)
    if should_hide:
        widget.pack_forget()
    else:
        widget.pack(fill="x", pady=spacing)
```
